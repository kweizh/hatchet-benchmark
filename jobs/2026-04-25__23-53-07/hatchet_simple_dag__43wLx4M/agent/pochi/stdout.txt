### Code Summary:
**worker.py**
```python
from hatchet_sdk import Hatchet

hatchet = Hatchet()

@hatchet.workflow(name="SimpleDAG")
class SimpleDAG:
    @hatchet.step()
    def step1(self, context):
        message = context.workflow_input().get("message")
        return {"data": message}

    @hatchet.step(parents=["step1"])
    def step2(self, context):
        step1_output = context.step_output("step1")
        data = step1_output.get("data")
        return {"result": data + " World!"}

if __name__ == "__main__":
    worker = hatchet.worker("simple-worker")
    worker.register_workflow(SimpleDAG())
    worker.start()
```

**trigger.py**
```python
from hatchet_sdk import Hatchet

hatchet = Hatchet()

def main():
    run = hatchet.client.admin.run_workflow("SimpleDAG", {"message": "Hello"})
    print(f"Triggered workflow run: {run.workflow_run_id}")
    result = run.result()
    step2_result = result.get("step2", {}).get("result")
    
    if step2_result:
        with open("/home/user/project/output.txt", "w") as f:
            f.write(step2_result)
        print(f"Result written to output.txt: {step2_result}")

if __name__ == "__main__":
    main()
```   
